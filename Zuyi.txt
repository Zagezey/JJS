local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- ScreenGui
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "AnimeKeyGUI"
screenGui.Parent = player:WaitForChild("PlayerGui")

-- Main Frame
local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 400, 0, 250)
frame.AnchorPoint = Vector2.new(0.5, 0.5)
frame.Position = UDim2.new(0.5, 0, 0.5, 0) -- Center of screen
frame.BackgroundColor3 = Color3.fromRGB(15, 15, 30)
frame.BorderSizePixel = 0
frame.Parent = screenGui
frame.Active = true
frame.Draggable = true

-- Rainbow LED Bar
local led = Instance.new("Frame")
led.Size = UDim2.new(1, 0, 0, 10)
led.Position = UDim2.new(0, 0, 0, 0)
led.BorderSizePixel = 0
led.Parent = frame

-- Animate Rainbow LED
spawn(function()
    local hue = 0
    while true do
        hue = (hue + 0.01) % 1
        led.BackgroundColor3 = Color3.fromHSV(hue, 1, 1)
        wait(0.03)
    end
end)

notification.BackgroundTransparency = 1
notification.TextColor3 = Color3.fromRGB(255,100,100)

notification.Parent = frame

-- Sound Setup
local sound = Instance.new("Sound")
sound.SoundId = "rbxassetid://116513148809589"
sound.Volume = 1
sound.Parent = frame

-- Get Key Button Function
getKeyBtn.MouseButton1Click:Connect(function()
    if setclipboard then
        setclipboard("https://discord.gg/wtKDQn7ZX") -- copy silently
        notification.TextColor3 = Color3.fromRGB(100,255,100)
        notification.Text = "üåà Key link copied!"
        sound:Play() -- play click sound
    else
        notification.TextColor3 = Color3.fromRGB(255,100,100)
        notification.Text = "Clipboard not supported!"
    end
end)

-- Check Key Button Function
checkKeyBtn.MouseButton1Click:Connect(function()
    local key = keyBox.Text
    if key == "MainZuyi" then
        notification.TextColor3 = Color3.fromRGB(100,255,100)
        notification.Text = "‚úÖ Key valid!"
        sound:Play() -- play correct key sound
        
        -- Load external script only if key is correct
        local success, err = pcall(function()
            loadstring(game:HttpGet("https://raw.githubusercontent.com/NotEnoughJack/LuaFluentDependancies/refs/heads/main/main.lua"))()
        end)
        if not success then
            warn("Failed to load script:", err)
        end
        
        -- Destroy the key system GUI
        screenGui:Destroy()
    else
        notification.TextColor3 = Color3.fromRGB(255,100,100)
        notification.Text = "‚ùå Invalid key!"
    end
end)
